public int singleNumber(int[] nums) {
    HashSet<Integer> hashSet = new HashSet<>();

    for(int i = 0; i < nums.length; i++) {
        int value = nums[i];

        if(!hashSet.contains(value)) hashSet.add(value);
        else hashSet.remove(value);
    }

    return result.intValue();
}
