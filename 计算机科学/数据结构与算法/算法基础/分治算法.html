          public static int binary_search(int[] a, int target) {
  int low = 0;
  int high = a.length - 1;
  int mid;

  while (low <= high) {
    mid = (high + low) / 2;

    if (a[mid] == target) {
      return mid;
    } else if (a[mid] < target) {
      low = mid;
    } else {
      high = mid;
    }
  }

  return -1;
}
